#Validation program: computes the classification for a data, and gives the evaluation metrics

Valid <- function(data,MMap,Mgbd,Egbd,MT,MTEB,alpha,wsd,COND,COND_only){

labs <- 0:nrow(Mgbd)

TR <- training(data,Mgbd,Egbd,MMap,MT,MTEB,alpha,wsd,gamma=NULL,cost=NULL,COND,COND_only)
MTc <- TR[[1]]
svm_mod_wrong <- TR[[2]]
CL <- classifier(data,MMap,MTc,wsd,svm_mod_wrong=NULL,COND,COND_only)
PERF <- Eval_CL(data, CL)
PERF_class <- Eval_per_class_CL(data, CL, labs)

return(list(PERF,PERF_class))
}
